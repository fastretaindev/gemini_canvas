// --- Logic: Data Sanitization ---


// 數值清洗：處理 7.253.00 (點號千分位) 與標準格式

function toNumber(v) {

  if (!v) return null;

  if (typeof v === 'number') return v;

  let s = v.trim();

  s = s.replace(/[$,]/g, ""); // 移除 $ 與 ,

  

  // 若有多個點，假設最後一個是小數點，其餘是千分位

  if ((s.match(/\./g) || []).length > 1) {

    s = s.replace(/\.(?=.*\.)/g, "");

  }

  

  const n = Number(s);

  return Number.isFinite(n) ? n : null;

}


// 座標行重組：核心邏輯

function groupTextItemsToLines(items) {

  const linesMap = new Map();

  const lineTolerance = 4;


  for (const item of items) {

    const transform = item.transform;

    const y = transform[5];

    const x = transform[4];

    const text = item.str || "";


    let key = [...linesMap.keys()].find(k => Math.abs(k - y) <= lineTolerance);

    if (key === undefined) key = y;


    const arr = linesMap.get(key) || [];

    arr.push({ x, text });

    linesMap.set(key, arr);

  }


  return Array.from(linesMap.entries())

    .sort((a, b) => b[0] - a[0])

    .map(([_, arr]) => {

      return arr.sort((a, b) => a.x - b.x).map(t => t.text).join(" ");

    });

}


// --- Logic: Advanced Parsers ---


// 金額抓取：支援寬字距標籤 (e.g. 銷 售 額) 與各式數值

function pickAmount(text, labelPattern) {

  const re = new RegExp(`${labelPattern}\\D*?([\\d,.]+)`);

  const m = text.match(re)?.[1] ?? null;

  return m ? toNumber(m) : null;

}


// 賣方區塊鎖定：避免抓到買方資料

function pickSellerBlock(text) {

  const m = text.match(/(?:營\s*業\s*人|賣\s*方)\s*[:：][\s\S]{0,300}/);

  return m?.[0] ?? "";

}


function parseSeller(text) {

  const block = pickSellerBlock(text);

  let seller_name = block.match(/[:：]\s*([^\n]+)/)?.[1]?.trim() || null;

  const seller_tax_id = block.match(/統\s*一\s*編\s*號\s*[:：.]?\s*(\d{8})/)?.[1] || null;

  return { seller_name, seller_tax_id };

}


// 稅別判定：支援跨行與分散關鍵字，新增 √ 符號支援

function pickTaxType(text) {

  const taxBlock = text.match(/營\s*業\s*稅[\s\S]{0,150}/)?.[0] ?? text;

  let taxable = "", zero = "", exempt = "";

  

  if (/應\s*稅\s*[:：]?\s*[VvXx✓√]/.test(taxBlock)) taxable = "V";

  if (/零\s*稅\s*率\s*[:：]?\s*[VvXx✓√]/.test(taxBlock)) zero = "V";

  if (/免\s*稅\s*[:：]?\s*[VvXx✓√]/.test(taxBlock)) exempt = "V";


  if (!taxable && !zero && !exempt) {

     if (/應\s*稅/.test(text) && !/零\s*稅/.test(text) && !/免\s*稅/.test(text)) taxable = "V";

  }

  return taxable ? "V" : (zero ? "Zero" : (exempt ? "Exempt" : "")); 

}


// --- Logic: Mode 1 - Header ---


function parseInvoiceHeader(lines) {

  const fullText = lines.join("\n");

  const mInv = fullText.match(/(?:發\s*票\s*號\s*碼|號\s*碼)[：:.]?\s*([A-Z]{2}[- ]?\d{8})/);

  const invoiceNumber = mInv ? mInv[1].replace(/[- ]/g, '') : "";

  const mFormat = fullText.match(/(?:格\s*式)[：:.]?\s*(\d{2})/);

  const format = mFormat ? mFormat[1] : "";

  const mDate = fullText.match(/(\d{4}[-/]\d{2}[-/]\d{2})/);

  const date = mDate ? mDate[1] : "";

  const { seller_name, seller_tax_id } = parseSeller(fullText);

  const taxStatus = pickTaxType(fullText);


  return {

    invoiceNumber, format, date,

    salesTotal: pickAmount(fullText, "銷\\s*售\\s*額\\s*合\\s*計")?.toString(),

    taxAmount: pickAmount(fullText, "(?:營\\s*業\\s*稅\\s*額|營\\s*業\\s*稅|稅\\s*額)")?.toString(),

    grandTotal: pickAmount(fullText, "總\\s*計")?.toString(),

    sellerId: seller_tax_id || "",

    sellerName: seller_name || "",

    taxTypeTaxable: taxStatus === "V" ? "V" : "",

    taxTypeZero: taxStatus === "Zero" ? "V" : "",

    taxTypeExempt: taxStatus === "Exempt" ? "V" : ""

  };

}


// --- Logic: Mode 2 - Details ---


function parseItemRow(line) {

  const normalized = line.replace(/\s+/g, " ").trim();

  const regexp = /^(.+)\s+([-]?[\d,.]+)\s+([-]?[$]?[\d,.]+)\s+([-]?[$]?[\d,.]+)(?:\s+(.*))?$/;

  const m = normalized.match(regexp);

  if (!m) return null;

  const cleanName = m[1].replace(/([\u4e00-\u9fa5])\s+([\u4e00-\u9fa5])/g, '$1$2').trim();


  return {

    description: cleanName,

    quantity: m[2],

    unitPrice: m[3].replace(/[$,]/g, ""), 

    amount: m[4].replace(/[$,]/g, ""), 

    remarks: m[5] ? m[5].trim() : ""

  };

}


function parseInvoiceDetails(lines, invoiceNumberVal, dateVal) {

  const results = [];

  let startIndex = -1;

  for (let i = 0; i < lines.length; i++) {

    if (lines[i].includes("品") && (lines[i].includes("數") || lines[i].includes("額"))) {

      startIndex = i + 1;

      break;

    }

  }


  if (startIndex !== -1) {

    for (let i = startIndex; i < lines.length; i++) {

      const line = lines[i].trim();

      if (!line) continue;

      if (line.match(/(總\s*計|合\s*計|銷\s*售\s*額|營\s*業\s*稅|賣\s*方|統\s*一\s*編\s*號)/)) break;

      const row = parseItemRow(line);

      if (row && !isNaN(Number(row.amount))) {

        results.push({ invoiceNumber: invoiceNumberVal, date: dateVal, ...row });

      }

    }

  }

  return results;

}


// --- Service: Orchestrator ---


const extractDataFromPdf = async (file, mode) => {

  const pdfjsLib = window.pdfjsLib;

  if (!pdfjsLib) throw new Error('PDF 核心尚未載入');

  const arrayBuffer = await file.arrayBuffer();

  const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;

  const extractedItems = [];


  for (let pageNum = 1; pageNum <= pdf.numPages; pageNum++) {

    const page = await pdf.getPage(pageNum);

    const textContent = await page.getTextContent();

    const lines = groupTextItemsToLines(textContent.items);

    const headerData = parseInvoiceHeader(lines);


    if (mode === 'header') {

      if (headerData.invoiceNumber || headerData.grandTotal) {

        extractedItems.push({ ...headerData, pageNumber: pageNum, rawLineString: lines });

      }

    } else {

      const details = parseInvoiceDetails(lines, headerData.invoiceNumber || "", headerData.date || "");

      if (details.length > 0) {

        details.forEach(d => extractedItems.push({ ...d, pageNumber: pageNum, rawLineString: lines }));

      } else if (headerData.invoiceNumber) {

         extractedItems.push({

             invoiceNumber: headerData.invoiceNumber, date: headerData.date,

             description: "", quantity: "", unitPrice: "", amount: "",

             status: 'warning', errorMessage: '未偵測到明細', pageNumber: pageNum, rawLineString: lines

         });

      }

    }

  }

  return extractedItems;

};

      )}

    </div>

  );

};
